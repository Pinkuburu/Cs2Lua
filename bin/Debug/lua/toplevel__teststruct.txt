require "cs2lua__namespaces"; 

TopLevel.TestStruct = {
	__new_object = function(...)
		return newobject(TopLevel.TestStruct, nil, nil, "ctor", nil, ...);
	end,
	cctor = function()
		TopLevel.TestStruct.__cctor();
	end,
	__cctor = function()
		if TopLevel.TestStruct.__cctor_called then 
			return;
		else
			TopLevel.TestStruct.__cctor_called = true;
		end;
	end,
	__define_class = function()
		Utility.Warn("TopLevel.TestStruct::__define_class begin");

		local class = TopLevel.TestStruct;
		local static_fields_build = function()
			local static_fields = {
				__cctor_called = false,
			};
			return static_fields;
		end;

		local instance_methods = {
			ctor = function(this)
				this:__ctor();
			end,
			__ctor = function(this)
				if this.__ctor_called then 
					return;
				else
					this.__ctor_called = true;
				end;
			end,
		};
		local instance_fields_build = function()
			local instance_fields = {
				A = 0,
				B = 0,
				C = 0,
				__ctor_called = false,
			};
			return instance_fields;
		end;

		local __defineclass_return = defineclass(nil, "TopLevel.TestStruct", "TestStruct", class, static_fields_build, instance_methods, instance_fields_build, true);
		Utility.Warn("TopLevel.TestStruct::__define_class end");
		return __defineclass_return;
	end,

	__interfaces = nil,
	__class_info = nil,
	__method_info = nil,
	__property_info = nil,
	__event_info = nil,
	__field_info = nil,
};


require "cs2lua__lualib";
require "cs2lua__externenums";
require "cs2lua__interfaces";
settempmetatable(TopLevel.TestStruct);
